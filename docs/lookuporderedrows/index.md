### Example 1: Basic Use

This example retrieves the top 3 customers from a Data Extension named `Customers` based on their `TotalOrders` in descending order, for customers located in 'New York'.

```html
%%[
VAR @deName, @numRows, @orderBy, @filterColumn, @filterValue
VAR @rows, @rowCount, @i, @customerName, @totalOrders

/* Define parameters for the LookupOrderedRows function */
SET @deName = "Customers"
SET @numRows = 3 /* Retrieve top 3 rows */
SET @orderBy = "TotalOrders DESC, CustomerName ASC" /* Sort by TotalOrders descending, then CustomerName ascending */
SET @filterColumn = "City"
SET @filterValue = "New York"

/* Perform the lookup */
SET @rows = LookupOrderedRows(@deName, @numRows, @orderBy, @filterColumn, @filterValue)

/* Check if any rows were returned */
IF NOT EMPTY(@rows) THEN
    SET @rowCount = RowCount(@rows)
    
    /* Output header */
    Output(Concat("<h3>Top ", @numRows, " Customers in ", @filterValue, ":</h3>"))
    Output("<ul>")

    /* Loop through the returned rows */
    FOR @i = 1 TO @rowCount DO
        SET @row = Row(@rows, @i)
        SET @customerName = Field(@row, "CustomerName")
        SET @totalOrders = Field(@row, "TotalOrders")
        
        Output(Concat("<li>", @customerName, " (Orders: ", @totalOrders, ")</li>"))
    NEXT @i
    
    Output("</ul>")
ELSE
    /* Handle case where no rows are found */
    Output(Concat("<p>No customers found in ", @filterValue, ".</p>"))
ENDIF
]%%
```

### Example 2: Advanced Scenario with Multiple Filters

This example demonstrates retrieving up to 5 recent orders for a specific `SubscriberKey` from an `OrderHistory` Data Extension, ordered by `OrderDate` descending, and further filtered by `OrderStatus`.

```html
%%[
VAR @deName, @numRows, @orderBy
VAR @subscriberKey, @orderStatus
VAR @rows, @rowCount, @i, @orderID, @orderDate, @itemCount

/* Define parameters for the LookupOrderedRows function */
SET @deName = "OrderHistory"
SET @numRows = 5 /* Retrieve up to 5 most recent orders */
SET @orderBy = "OrderDate DESC" /* Sort by OrderDate in descending order */

/* Define filter criteria */
SET @subscriberKey = AttributeValue("SubscriberKey") /* Assuming SubscriberKey is available in the send context */
SET @orderStatus = "Completed"

/* Perform the lookup with multiple filter pairs */
SET @rows = LookupOrderedRows(
    @deName,
    @numRows,
    @orderBy,
    "SubscriberKey", @subscriberKey,
    "OrderStatus", @orderStatus
)

/* Check if any rows were returned */
IF NOT EMPTY(@rows) THEN
    SET @rowCount = RowCount(@rows)
    
    /* Output header */
    Output(Concat("<h3>Recent Completed Orders for Subscriber: ", @subscriberKey, ":</h3>"))
    Output("<table border='1'>")
    Output("<tr><th>Order ID</th><th>Order Date</th><th>Items</th></tr>")

    /* Loop through the returned rows */
    FOR @i = 1 TO @rowCount DO
        SET @row = Row(@rows, @i)
        SET @orderID = Field(@row, "OrderID")
        SET @orderDate = Field(@row, "OrderDate")
        SET @itemCount = Field(@row, "ItemCount")
        
        Output(Concat("<tr><td>", @orderID, "</td><td>", FormatDate(@orderDate, "MMM D, YYYY"), "</td><td>", @itemCount, "</td></tr>"))
    NEXT @i
    
    Output("</table>")
ELSE
    /* Handle case where no rows are found */
    Output(Concat("<p>No completed orders found for Subscriber: ", @subscriberKey, ".</p>"))
ENDIF
]%%
```
